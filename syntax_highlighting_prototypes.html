# Sample Python file
import json
from datetime import datetime

class TaskManager:
    def __init__(self):
        self.tasks = []
    
    def add_task(self, title, priority="medium"):
        task = {
            "id": len(self.tasks) + 1,
            "title": title,
            "priority": priority,
            "created_at": datetime.now().isoformat(),
            "completed": False
        }
        self.tasks.append(task)
        return task
    
    def complete_task(self, task_id):
        for task in self.tasks:
            if task["id"] == task_id:
                task["completed"] = True
                return True
        return False

# Usage
manager = TaskManager()
manager.add_task("Research syntax highlighting", "high")
print(json.dumps(manager.tasks, indent=2))

// Sample TypeScript file
interface FileViewerConfig {
    theme: 'light' | 'dark';
    showLineNumbers: boolean;
    enableSyntaxHighlighting: boolean;
    supportedExtensions: string[];
}

class FileViewer {
    private config: FileViewerConfig;
    private container: HTMLElement;

    constructor(container: HTMLElement, config: Partial<FileViewerConfig> = {}) {
        this.container = container;
        this.config = {
            theme: 'light',
            showLineNumbers: true,
            enableSyntaxHighlighting: true,
            supportedExtensions: ['.js', '.ts', '.py', '.html', '.css', '.json'],
            ...config
        };
    }

    public renderFile(content: string, extension: string): void {
        if (!this.config.supportedExtensions.includes(extension)) {
            throw new Error(`Unsupported file extension: ${extension}`);
        }

        const highlighted = this.config.enableSyntaxHighlighting 
            ? this.highlightSyntax(content, extension)
            : content;

        this.container.innerHTML = this.wrapContent(highlighted);
    }

    private highlightSyntax(content: string, extension: string): string {
        // Implementation would depend on chosen highlighting library
        return content;
    }

    private wrapContent(content: string): string {
        return `<pre><code>${content}</code></pre>`;
    }
}

// Usage
const viewer = new FileViewer(
    document.getElementById('file-container')!,
    { theme: 'dark', showLineNumbers: true }
);

4. JavaScript File Extension Mapping Function

// File extension to language mapping function
function getLanguageFromExtension(filePath) {
    const extensionMap = {
        // JavaScript & TypeScript
        '.js': 'javascript',
        '.jsx': 'jsx',
        '.ts': 'typescript',
        '.tsx': 'tsx',
        
        // Web technologies
        '.html': 'html',
        '.htm': 'html',
        '.css': 'css',
        '.json': 'json',
        
        // Programming languages
        '.py': 'python',
        '.java': 'java',
        '.cpp': 'cpp',
        '.c': 'c',
        '.php': 'php',
        '.rb': 'ruby',
        '.go': 'go',
        '.rs': 'rust',
        '.swift': 'swift',
        '.kt': 'kotlin',
        
        // Markup & Config
        '.md': 'markdown',
        '.xml': 'xml',
        '.yaml': 'yaml',
        '.yml': 'yaml',
        '.toml': 'toml',
        '.ini': 'ini',
        
        // Database
        '.sql': 'sql',
        
        // Shell & Scripts
        '.sh': 'bash',
        '.bash': 'bash',
        '.zsh': 'zsh',
        '.ps1': 'powershell',
        '.bat': 'batch',
        
        // Data formats
        '.csv': 'csv',
        '.tsv': 'csv'
    };
    
    const extension = filePath.toLowerCase().substring(filePath.lastIndexOf('.'));
    return extensionMap[extension] || 'text';
}

// Usage example
console.log(getLanguageFromExtension('app.py'));        // 'python'
console.log(getLanguageFromExtension('style.css'));     // 'css'
console.log(getLanguageFromExtension('config.json'));   // 'json'
console.log(getLanguageFromExtension('README.md'));     // 'markdown'
